 CREATE DATABASE SMMDB;

//USER TABLE
USE SMMDB;
CREATE TABLE USER(USER_ID INT PRIMARY KEY
,NAME VARCHAR(30)
,PASSWORD VARCHAR(30)
,TYPE VARCHAR(30));

//SHOP TABLE
USE SMMDB;
CREATE TABLE SHOP(SHOPID INT PRIMARY KEY
,LEASESTATUS VARCHAR(30)
,SHOP_CATEGORY VARCHAR(30)
,SHOPNAME VARCHAR(30)
,SHOPSTATUS VARCHAR(30),
MALLID INT);

//CUSTOMER
USE SMMDB
CREATE TABLE CUSTOMER(CUSTOMERID INT NOT NULL PRIMARY KEY
,EMAIL VARCHAR(30)
,NAME VARCHAR(30)
,PHONE VARCHAR(30)
,SHOPID VARCHAR(30)
,USERID VARCHAR(30));

//EMPLOYEE
USE SMMDB;
CREATE TABLE EMPLOYEE(EMP_ID INT PRIMARY KEY 
,ADDRESS VARCHAR(50)
,DOB DATE
,DESIGNATION VARCHAR(20)
,NAME VARCHAR(30)
,SALARY FLOAT
,SHOPID INT);

//ITEM
CREATE TABLE ITEM(ITEMID INT PRIMARY KEY 
,NAME VARCHAR 
,PRICE FLOAT
,MAN_DATE DATE
,EXP_DATE DATE
,CATEGORY VARCHAR(30)
,SHOPID INT);


//MALL
CREATE TABLE MALL(MALLID INT,
CATEGORY VARCHAR(30),
LOCATION VARCHAR(30),
NAME VARCHAR(30),
MALL_ADMINID INT);

//MALL ADMIN
CREATE TABLE MALLADMIN(ADMINID INT PRIMARY KEY,
NAME VARCHAR(30),
PASSWORD VARCHAR(30),
PHONE NUMERIC,
USERID INT);


//ORDERDETAIL
CREATE TABLE ORDERDETAIL(ORDERID INT PRIMARY KEY,
DATEOFPURCHASE DATE,
PAYMENTMODE VARCHAR(20),
TOTAL FLOAT,
CUSTOMERID INT, 
SHOPID INT);
